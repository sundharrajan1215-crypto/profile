#include <iostream>
#include <fstream>
#include <string>

using namespace std;

class Library
{
private:
    int bookID;
    string title;
    string author;
    bool issued;

public:

    void addBook()
    {
        cout << "\nEnter Book ID : ";
        cin >> bookID;

        cin.ignore();

        cout << "Enter Book Title : ";
        getline(cin, title);

        cout << "Enter Author Name : ";
        getline(cin, author);

        issued = false;

        ofstream file("library.txt", ios::app);

        file << bookID << endl;
        file << title << endl;
        file << author << endl;
        file << issued << endl;

        file.close();

        cout << "\nBook Added Successfully!\n";
    }

    void displayBooks()
    {
        ifstream file("library.txt");

        cout << "\n========== BOOK LIST ==========\n";

        while(file >> bookID)
        {
            file.ignore();

            getline(file, title);
            getline(file, author);

            file >> issued;

            cout << "\nBook ID : " << bookID;
            cout << "\nTitle   : " << title;
            cout << "\nAuthor  : " << author;

            if(issued)
                cout << "\nStatus  : Issued\n";
            else
                cout << "\nStatus  : Available\n";
        }

        file.close();
    }

    void issueBook()
    {
        int id;
        bool found = false;

        cout << "\nEnter Book ID to Issue : ";
        cin >> id;

        ifstream file("library.txt");
        ofstream temp("temp.txt");

        while(file >> bookID)
        {
            file.ignore();

            getline(file, title);
            getline(file, author);

            file >> issued;

            if(bookID == id)
            {
                found = true;

                if(!issued)
                {
                    issued = true;
                    cout << "\nBook Issued Successfully!\n";
                }
                else
                {
                    cout << "\nBook Already Issued!\n";
                }
            }

            temp << bookID << endl;
            temp << title << endl;
            temp << author << endl;
            temp << issued << endl;
        }

        file.close();
        temp.close();

        remove("library.txt");
        rename("temp.txt", "library.txt");

        if(!found)
        {
            cout << "\nBook Not Found!\n";
        }
    }

    void returnBook()
    {
        int id;
        bool found = false;

        cout << "\nEnter Book ID to Return : ";
        cin >> id;

        ifstream file("library.txt");
        ofstream temp("temp.txt");

        while(file >> bookID)
        {
            file.ignore();

            getline(file, title);
            getline(file, author);

            file >> issued;

            if(bookID == id)
            {
                found = true;

                if(issued)
                {
                    issued = false;
                    cout << "\nBook Returned Successfully!\n";
                }
                else
                {
                    cout << "\nBook Was Not Issued!\n";
                }
            }

            temp << bookID << endl;
            temp << title << endl;
            temp << author << endl;
            temp << issued << endl;
        }

        file.close();
        temp.close();

        remove("library.txt");
        rename("temp.txt", "library.txt");

        if(!found)
        {
            cout << "\nBook Not Found!\n";
        }
    }

    void searchBook()
    {
        int choice;
        bool found = false;

        cout << "\nSearch By:";
        cout << "\n1. Title";
        cout << "\n2. Author";

        cout << "\nEnter Choice : ";
        cin >> choice;

        cin.ignore();

        string search;

        if(choice == 1)
        {
            cout << "\nEnter Title : ";
            getline(cin, search);
        }
        else if(choice == 2)
        {
            cout << "\nEnter Author : ";
            getline(cin, search);
        }

        ifstream file("library.txt");

        while(file >> bookID)
        {
            file.ignore();

            getline(file, title);
            getline(file, author);

            file >> issued;

            if((choice == 1 && title == search) ||
               (choice == 2 && author == search))
            {
                cout << "\nBook ID : " << bookID;
                cout << "\nTitle   : " << title;
                cout << "\nAuthor  : " << author;

                if(issued)
                    cout << "\nStatus  : Issued\n";
                else
                    cout << "\nStatus  : Available\n";

                found = true;
            }
        }

        file.close();

        if(!found)
        {
            cout << "\nNo Matching Book Found!\n";
        }
    }
};

int main()
{
    Library l;
    int choice;

    do
    {
        cout << "\n========== LIBRARY MANAGEMENT SYSTEM ==========\n";

        cout << "1. Add Book\n";
        cout << "2. Display Books\n";
        cout << "3. Issue Book\n";
        cout << "4. Return Book\n";
        cout << "5. Search Book\n";
        cout << "6. Exit\n";

        cout << "\nEnter Your Choice : ";
        cin >> choice;

        switch(choice)
        {
            case 1:
                l.addBook();
                break;

            case 2:
                l.displayBooks();
                break;

            case 3:
                l.issueBook();
                break;

            case 4:
                l.returnBook();
                break;

            case 5:
                l.searchBook();
                break;

            case 6:
                cout << "\nThank You!\n";
                break;

            default:
                cout << "\nInvalid Choice!\n";
        }

    } while(choice != 6);

    return 0;
}
